## 动态规划

> 力扣题目5：给定一个字符串 s，找到 s 中最长的回文子串。你可以假设 s 的最大长度为 1000。

```python
class Solution:
    def longestPalindrome(self, s: str) -> str:
        # 动态规划，记忆化搜索
        dp = [[-1 for _ in range(len(s))] for __ in range(len(s))]

        def get_dp(i, j):
            if dp[i][j] >= 0:
                return dp[i][j]
            if i==j:
                dp[i][j] = 1
            elif i+1==j:
                dp[i][j] = 2 if s[i]==s[j] else 0
            else:
                if s[i]==s[j] and get_dp(i+1,j-1)>0:
                    dp[i][j] = j-i+1
                else:
                    dp[i][j] = 0
            return dp[i][j]

        maxlen = 0
        result = ""
        for j in range(len(s)):
            for i in range(j+1):
                if get_dp(i,j) > maxlen:
                    maxlen = get_dp(i,j)
                    result = s[i:j+1]

        return result
```

class Solution:
    def longestPalindrome(self, s: str) -> str:
        # 中间拓展
        def get_max(i):
            max_len = 1
            max_string = s[i]
            d = 1
            while i-d>=0 and i+d<len(s) and s[i-d]==s[i+d]:
                d += 1
            if 2*d-1 > max_len:
                max_len = 2*d - 1
                max_string = s[i-d+1:i+d]
            d = 0
            while i-1-d>=0 and i+d<len(s) and s[i-d-1]==s[i+d]:
                d += 1
            if 2*d > max_len:
                max_len = 2*d
                max_string = s[i-d:i+d]
            return max_string

        result = ""
        for i in range(len(s)):
            temp = get_max(i)
            if len(temp) > len(result):
                result = temp
        return result
